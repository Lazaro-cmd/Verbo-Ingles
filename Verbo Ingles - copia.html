<script>

// Ejemplo de datos simples. Personaliza o expande según quieras.

const verbs = [

{ base: "be", past: "was/were", part: "been", meaning: "ser/estar", audio: "" },

{ base: "become", past: "became", part: "become", meaning: "convertirse", audio: "" },

{ base: "begin", past: "began", part: "begun", meaning: "empezar", audio: "" },

{ base: "break", past: "broke", part: "broken", meaning: "romper", audio: "" },

{ base: "bring", past: "brought", part: "brought", meaning: "traer", audio: "" },

// ... añade aquí más verbos ...

];



let current = 0;

let flipped = false;

let studyMode = true;

let shown = Array(verbs.length).fill(false);



function updateCard() {

flipped = false;

document.getElementById('flipCard').classList.remove('flipped');

let v = verbs[current];

document.getElementById("front").innerHTML = `<b>${v.base}</b><br>(${v.meaning})`;

document.getElementById("back").innerHTML = `

<b>${v.base}</b> - <b>${v.past}</b> - <b>${v.part}</b>

<br>[translate:${v.meaning}]

<br>

`;

updateStats();

}



function updateStats() {

let studied = shown.filter(Boolean).length;

document.getElementById("stats").textContent = `Vistos: ${studied} de ${verbs.length}`;

}



document.getElementById("flip").onclick = () => {

flipped = !flipped;

document.getElementById('flipCard').classList.toggle('flipped');

};

document.getElementById("next").onclick = () => {

current = (current + 1) % verbs.length;

shown[current] = true;

updateCard();

};

document.getElementById("prev").onclick = () => {

current = (current - 1 + verbs.length) % verbs.length;

shown[current] = true;

updateCard();

};

document.getElementById("reset").onclick = () => {

current = 0; shown = Array(verbs.length).fill(false); updateCard();

};

document.getElementById("random").onclick = () => {

current = Math.floor(Math.random()*verbs.length); shown[current] = true; updateCard();

};

document.getElementById("study").onclick = () => {

studyMode = true;

document.getElementById("study").classList.add("selected");

document.getElementById("practice").classList.remove("selected");

};

document.getElementById("practice").onclick = () => {

studyMode = false;

document.getElementById("practice").classList.add("selected");

document.getElementById("study").classList.remove("selected");

};



document.getElementById("flipCard").onclick = () => {

document.getElementById("flip").click();

};



document.getElementById("search").oninput = function() {

const val = this.value.trim().toLowerCase();

if (!val) return updateCard();

let found = verbs.findIndex(

v => v.base.toLowerCase().includes(val) ||

v.past.toLowerCase().includes(val) ||

v.part.toLowerCase().includes(val) ||

v.meaning.toLowerCase().includes(val)

);

if (found !== -1) {

current = found;

shown[current] = true;

updateCard();

}

};



updateCard();

</script>
